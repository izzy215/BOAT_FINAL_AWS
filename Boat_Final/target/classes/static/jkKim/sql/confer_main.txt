DROP table CONFER_MAIN CASCADE CONSTRAINTS PURGE;

CREATE TABLE CONFER_MAIN(
	ID VARCHAR2(30),
	RENTAL VARCHAR2(50),  --대여품이름
	START_TIME VARCHAR2(50),
	END_TIME VARCHAR2(50),
	CONTENT VARCHAR2(150),
	START_T VARCHAR2(50), --view단출력용
	END_T VARCHAR2(50),
	STATUS VARCHAR2(10) DEFAULT 0, --처리중인상태
	MEMO VARCHAR2(200) DEFAULT '',--사유출력용
	ABC VARCHAR2(50) --입력시간
	);
